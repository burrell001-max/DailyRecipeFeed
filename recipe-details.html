function renderRecipeDetails() { const container = document.getElementById('content-container'); const catalog = (typeof recipeCatalog !== 'undefined') ? recipeCatalog : []; // 1. Grab the "id" string out of the browser URL bar (?id=ramen-salad) const urlParams = new URLSearchParams(window.location.search); const recipeId = urlParams.get('id'); if (!recipeId || catalog.length === 0) { showErrorLayout(container); return; } // 2. Locate the corresponding item inside the recipes array const item = catalog.find(r => r.id === recipeId || (r.url && r.url.includes(`id=${recipeId}`))); if (!item) { showErrorLayout(container); return; } // Update Browser Title Bar dynamically document.title = `${item.title} | DailyRecipeFeed`; // 3. Inject full layout structure safely with clean default fallbacks container.innerHTML = ` ← Back to Recipes
${item.category || 'Recipe'}

${item.title}

★★★★★ (${item.reviews || '24'} Reviews)
${item.description ? `

"${item.description}"

` : ''}

Ingredients

    ${item.ingredients && item.ingredients.length > 0 ? item.ingredients.map(ing => `
  • ${ing}
    • `).join('') : '
  • Ingredients coming soon!
    • ' }

Directions

    ${item.directions && item.directions.length > 0 ? item.directions.map(step => `
  1. ${step}
    2. `).join('') : '
  2. Directions coming soon!
    3. ' }
`; } function showErrorLayout(element) { element.innerHTML = ` ← Back to Home

Recipe Not Found

We couldn't track down that specific formula. Double-check your pathing details inside the data configurations file.

`; } window.onload = renderRecipeDetails;
